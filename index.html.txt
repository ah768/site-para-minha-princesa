<script>
function mostrarParte2(){
  document.getElementById('parte1').style.display='none';
  document.getElementById('parte2').style.display='flex';

  const input = document.getElementById('uploadFotos');
  const container = document.getElementById('imagens');
  const mensagemFinal = document.getElementById('mensagem-final');

  input.addEventListener('change', function(e){
    const arquivos = e.target.files;
    container.innerHTML = '';
    let carregadas = 0;

    Array.from(arquivos).forEach(file => {
      const img = document.createElement('img');
      img.src = URL.createObjectURL(file);
      container.appendChild(img);

      img.onload = () => {
        carregadas++;
        if(carregadas === arquivos.length){
          mensagemFinal.style.display = 'block';
        }
      }
    });
  });
}
</script>
